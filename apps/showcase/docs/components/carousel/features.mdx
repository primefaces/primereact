## Usage

```tsx
import { Carousel } from 'primereact/carousel';
```

```tsx
<Carousel>
    <Carousel.Content>
        <Carousel.Item></Carousel.Item>
    </Carousel.Content>
    <Carousel.Indicators />
    <Carousel.Prev></Carousel.Prev>
    <Carousel.Next></Carousel.Next>
</Carousel>
```
