---
title: Tooltip
description: Tooltip is a component that displays a tooltip when the user hovers over an element.
component: tooltip
---

## Usage

```tsx
import { Tooltip } from 'primereact/tooltip';
```

```tsx
<Tooltip.Group>
    <Tooltip>
        <Tooltip.Trigger></Tooltip.Trigger>
        <Tooltip.Portal>
            <Tooltip.Content>
                <Tooltip.Arrow />
            </Tooltip.Content>
        </Tooltip.Portal>
    </Tooltip>
</Tooltip.Group>
```
