## Usage

```tsx
import { Breadcrumb } from 'primereact/breadcrumb';
```

```tsx
<Breadcrumb>
    <Breadcrumb.List>
        <Breadcrumb.Item />
        <Breadcrumb.Separator />
    </Breadcrumb.List>
</Breadcrumb>
```
