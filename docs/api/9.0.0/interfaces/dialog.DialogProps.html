Properties

Optional appendTo

appendTo?: null | HTMLElement | "self"

DOM element instance where the overlay panel should be mounted. Valid values are any DOM Element and 'self'. The self value is used to render a component where it is located.

Default Value

document.body

Optional ariaCloseIconLabel

ariaCloseIconLabel?: string

Defines a string that labels the close icon.

Optional baseZIndex

baseZIndex?: number

Base zIndex value to use in layering.

Default Value

0

Optional blockScroll

blockScroll?: boolean

Whether background scroll should be blocked when dialog is visible.

Default Value

false

Optional breakpoints

breakpoints?: DialogBreakpoints

Object literal to define widths per screen size.

Optional Readonly children

children?: ReactNode

Used to get the child elements of the component.

Optional className

className?: string

Style class of the component.

Optional closable

closable?: boolean

Adds a close icon to the header to hide the dialog.

Default Value

true

Optional closeOnEscape

closeOnEscape?: boolean

Specifies if pressing escape key should hide the dialog.

Default Value

true

Optional contentClassName

contentClassName?: string

Style class of the content section.

Optional contentStyle

contentStyle?: CSSProperties

Style of the content section.

Optional dismissableMask

dismissableMask?: boolean

Specifies if clicking the modal background should hide the dialog.

Default Value

false

Optional draggable

draggable?: boolean

Enables dragging to change the position using header.

Default Value

true

Optional focusOnShow

focusOnShow?: boolean

When enabled, first button receives focus on show.

Default Value

true

Optional footer

footer?: ReactNode | ((props: DialogProps) => ReactNode)

Footer content of the dialog.

Optional header

header?: ReactNode | ((props: DialogProps) => ReactNode)

Title content of the dialog.

Optional headerClassName

headerClassName?: string

Style class of the header section.

Optional headerStyle

headerStyle?: CSSProperties

Style of the header section.

Optional icons

icons?: ReactNode | ((props: DialogProps) => ReactNode)

Custom icons template for the header.

Optional id

id?: string

Unique identifier of the element.

Optional keepInViewport

keepInViewport?: boolean

Keeps dialog in the viewport.

Default Value

true

Optional maskClassName

maskClassName?: string

Style class of the mask.

Optional maskStyle

maskStyle?: CSSProperties

Inline style of the mask.

Optional maximizable

maximizable?: boolean

Whether the dialog can be displayed full screen.

Default Value

false

Optional maximized

maximized?: boolean

When enabled, the dialog is initially displayed full screen.

Default Value

false

Optional minX

minX?: number

Minimum value for the left coordinate of dialog in dragging.

Default Value

0

Optional minY

minY?: number

Minimum value for the top coordinate of dialog in dragging.

Default Value

0

Optional modal

modal?: boolean

Defines if background should be blocked when dialog is displayed.

Default Value

true

Optional position

position?: "center" | "left" | "top" | "bottom" | "right" | "bottom-right" | "bottom-left" | "top-right" | "top-left"

Position of the dialog, options are "center", "top", "bottom", "left", "right", "top-left", "top-right", "bottom-left" or "bottom-right".

Default Value

center

Optional resizable

resizable?: boolean

Enables resizing of the content.

Default Value

true

Optional rtl

rtl?: boolean

When enabled dialog is displayed in RTL direction.

Default Value

false

Optional showHeader

showHeader?: boolean

Whether to show the header or not.

Default Value

true

Optional style

style?: CSSProperties

Inline style of the component.

Optional transitionOptions

transitionOptions?: CSSTransitionProps<undefined>

The properties of CSSTransition can be customized, except for "nodeRef" and "in" properties.

Optional visible

visible?: boolean

Specifies the visibility of the dialog.

Default Value

false
