Properties

Optional accept

accept?: string

Pattern to restrict the allowed file types such as "image/*".

Default Value

false

Optional auto

auto?: boolean

When enabled, upload begins automatically after selection is completed.

Default Value

false

Optional cancelLabel

cancelLabel?: string

Label of the cancel button. Defaults to global value in Locale configuration.

Optional cancelOptions

cancelOptions?: FileUploadOptions

Options used to customize the cancel button. These options have "label", "icon", "className" and "style" properties.

Optional Readonly children

children?: ReactNode

Used to get the child elements of the component.

Optional chooseLabel

chooseLabel?: string

Label of the choose button. Defaults to global value in Locale configuration.

Optional chooseOptions

chooseOptions?: FileUploadOptions

Options used to customize the choose button. These options have "label", "icon", "className" and "style" properties.

Optional className

className?: string

Style class of the component.

Optional contentClassName

contentClassName?: string

Style class of the content.

Optional contentStyle

contentStyle?: CSSProperties

Inline style of the content.

Optional customUpload

customUpload?: boolean

Whether to use the default upload or a manual implementation defined in uploadHandler callback.

Default Value

false

Optional disabled

disabled?: boolean

Disables the upload functionality.

Default Value

false

Optional emptyTemplate

emptyTemplate?: ReactNode | ((props: FileUploadProps) => ReactNode)

Custom template of empty content in the container.

Optional headerClassName

headerClassName?: string

Custom template of the header.

Optional headerStyle

headerStyle?: CSSProperties

Inline style of the header.

Optional headerTemplate

headerTemplate?: ReactNode | ((options: FileUploadHeaderTemplateOptions) => ReactNode)

Custom template of the header.

Optional id

id?: string

Unique identifier of the element.

Optional invalidFileSizeMessageDetail

invalidFileSizeMessageDetail?: string

Detail message of the invalid fize size.

Default Value

"Maximum upload size is."

Optional invalidFileSizeMessageSummary

invalidFileSizeMessageSummary?: string

Summary message of the invalid fize size.

Default Value

Invalid file size

Optional itemTemplate

itemTemplate?: ReactNode | ((file: object, options: ItemTemplateOptions) => ReactNode)

Custom template of each item content in the container.

Optional maxFileSize

maxFileSize?: number

Maximum file size allowed in bytes.

Optional mode

mode?: "basic" | "advanced"

Defines the UI of the component, possible values are "advanced" and "basic".

Default Value

advanced

Optional multiple

multiple?: boolean

Used to select multiple files at once from file dialog.

Default Value

false

Optional name

name?: string

Name of the request parameter to identify the files at backend.

Optional previewWidth

previewWidth?: number

Width of the image thumbnail in pixels.

Default Value

50

Optional progressBarTemplate

progressBarTemplate?: ReactNode | ((props: FileUploadProps) => ReactNode)

Custom template of progressBar content in the container.

Optional style

style?: CSSProperties

Inline style of the component.

Optional uploadLabel

uploadLabel?: string

Label of the upload button. Defaults to global value in Locale configuration.

Optional uploadOptions

uploadOptions?: FileUploadOptions

Options used to customize the upload button. These options have "label", "icon", "className" and "style" properties.

Optional url

url?: string

Remote url to upload the files.

Optional withCredentials

withCredentials?: boolean

Cross-site Access-Control requests should be made using credentials such as cookies, authorization headers or TLS client certificates.

Default Value

false
